#include <bits/stdc++.h>
using namespace std;
int n, a[200005], dp[200005], mx[200005];
int main () {
    freopen("remove.in",  "r", stdin);
    freopen("remove.out", "w", stdout);
	scanf("%d", &n);
    for (int i  = 1; i <= n; i++) {
    	   scanf("%d", &a[i]);
           dp[i] = 0;
           mx[i] = -2e5;
    }
    for (int i = 1; i <= n; i++) {
		dp[i] = max(dp[i-1], mx[a[i]] + i);
		mx[a[i]] = max(mx[a[i]], dp[i - 1] - (i - 1));
    }
    cout << dp[n] << endl;
	return 0;
}
