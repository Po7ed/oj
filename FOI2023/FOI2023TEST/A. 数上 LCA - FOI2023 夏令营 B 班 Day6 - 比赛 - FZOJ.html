下面附上线性筛的代码：

const int N=1e7+3;
bool np[N];
int pri[N/10],cnt;
void init(int n){
    for(int i=2;i<=n;++i){
        if(!np[i])pri[++cnt]=i;
	    for(int j=1;j<=cnt&&i*pri[j]<=n;++j){
	        np[i*pri[j]]=1;
	        if(!(i%pri[j]))break;
	    }
    }
}
